# armspan_height_analysis.py
# 📐 Armspan vs Height Analysis
# This script tests Leonardo da Vinci’s claim that a person’s armspan is approximately equal to their height.

import matplotlib.pyplot as plt
import numpy as np

# ---------------------------------------------
# 📊 Dataset: Armspan and Height of 8 individuals
# ---------------------------------------------
armspan = [172, 158, 165, 176, 172, 175, 157, 153]
height = [175, 157, 165, 177, 170, 170, 160, 157]

# ---------------------------------------------
# 🔹 (a) Scatterplot of Armspan vs Height
# ---------------------------------------------
plt.figure(figsize=(6, 6))
plt.scatter(armspan, height, color='blue', edgecolors='black', s=100)
plt.xlabel('Armspan (cm)')
plt.ylabel('Height (cm)')
plt.title('Armspan vs Height Scatterplot')
plt.grid(True)
plt.gca().set_aspect('equal')  # Equal scale for better visual comparison
plt.xlim(150, 180)
plt.ylim(150, 180)
plt.tight_layout()
plt.show()

# ---------------------------------------------
# 🔹 (b) Expected Slope if Armspan = Height
# ---------------------------------------------
print("🔹 Theoretical slope if armspan equals height: 1")

# ---------------------------------------------
# 🔹 (c) Linear Regression: Fit line Height = b * Armspan + a
# ---------------------------------------------
x = np.array(armspan)
y = np.array(height)

# Fit the line using numpy polyfit
b, a = np.polyfit(x, y, 1)

# Display regression results
print(f"\n🔹 Regression equation:")
print(f"   Height = {b:.3f} * Armspan + {a:.3f}")
print(f"   (Slope = {b:.3f})")

# ---------------------------------------------
# 🔹 (d) Predict Height for Armspan = 157 cm
# ---------------------------------------------
armspan_input = 157
predicted_height = b * armspan_input + a

print(f"\n🔹 Predicted height for armspan of 157 cm: {predicted_height:.2f} cm")
